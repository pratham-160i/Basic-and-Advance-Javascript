Homework Exercises

Exercise 1: Function with Rest Parameter

function joinNames(separator, ...names) {
  return names.join(separator);
}

console.log(joinNames(", ", "Priya", "Aarav", "Riya"));
// "Priya, Aarav, Riya"

Exercise 2: Merge Settings

const defaultSettings = { 
  theme: "light", 
  lang: "en", 
  notifications: true 
};

const userSettings = { 
  theme: "dark", 
  fontSize: 16 
};

function applyPrefs(defaults, prefs) {
  return { ...defaults, ...prefs };
}

const final = applyPrefs(defaultSettings, userSettings);
console.log(final);

Exercise 3: Extract Data

const student = {
  id: 101,
  name: "Gokul",
  grades: { math: 95, english: 88 },
  enrolled: true
};

// Extract what you need
const { name, grades: { math }, enrolled } = student;

console.log(`${name} scored ${math} and is enrolled: ${enrolled}`);
