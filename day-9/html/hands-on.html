Task 2: Array + Object Destructuring

const scores = [88, 75, 92, 60, 45];
const [top1, second, ...others] = scores;

console.log(top1);    // 88
console.log(second);  // 75
console.log(others);  // [92, 60, 45]

const user = {
  name: "Anaya",
  age: 21,
  address: {
    city: "Jaipur",
    pincode: "302001"
  }
};

const { name, age: userAge, address: { city } } = user;
console.log(name);     // "Anaya"
console.log(userAge);  // 21
console.log(city);     // "Jaipur"

Task 3: Spread Operator

const defaults = { theme: "light", lang: "en", notifications: true };
const userPrefs = { theme: "dark", fontSize: 16 };

const merged = { ...defaults, ...userPrefs };
console.log(merged);
// { theme: "dark", lang: "en", notifications: true, fontSize: 16 }
