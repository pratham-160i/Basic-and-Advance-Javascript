Exercise 2: Temperature Converter

function celsiusToFahrenheit(celsius) {
  return (celsius * 9/5) + 32;
}

function fahrenheitToCelsius(fahrenheit) {
  return (fahrenheit - 32) * 5/9;
}

console.log(celsiusToFahrenheit(25));   // 77
console.log(fahrenheitToCelsius(68));   // 20

Exercise 3: Age Category

function getAgeCategory(age) {
  if (age < 13) return "Child";
  if (age < 18) return "Teenager";
  if (age < 60) return "Adult";
  return "Senior";
}

console.log(getAgeCategory(10));   // Child
console.log(getAgeCategory(15));   // Teenager
console.log(getAgeCategory(25));   // Adult
console.log(getAgeCategory(70));   // Senior
