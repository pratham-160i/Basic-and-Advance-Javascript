Exercise 1: Dynamic List

Click the button below to dynamically create and add items to the list:

Code:

let list = document.querySelector("#dynamic-list");

let items = ["Learn JavaScript", "Build Projects", "Master DOM"];

items.forEach(function(item) {
  let li = document.createElement("li");
  li.textContent = item;
  list.appendChild(li);
});

Exercise 2: Toggle Visibility

Click the button to toggle the visibility of the box below:

This box can be toggled on and off!

Hidden element will display: none

Code:

function toggleElement(selector) {
  let elem = document.querySelector(selector);
  if (elem) {
    if (elem.style.display === "none") {
      elem.style.display = "block";
    } else {
      elem.style.display = "none";
    }
  }
}

// Usage
toggleElement(".box");

Exercise 3: Update Multiple Elements

Click any button below to style all buttons:

Code:

// Change all buttons
let buttons = document.querySelectorAll(".style-me");
buttons.forEach(function(btn) {
  btn.style.backgroundColor = "#007bff";
  btn.style.color = "white";
  btn.style.padding = "10px 20px";
  btn.style.border = "none";
  btn.style.borderRadius = "4px";
  btn.style.cursor = "pointer";
});
