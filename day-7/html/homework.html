Homework Exercises

Exercise 1: Calculate Average

let marks = [75, 82, 90, 68, 88];
let sum = 0;

for (let mark of marks) {
  sum += mark;
}

let average = sum / marks.length;
console.log("Average: " + average);  // 80.6

Exercise 2: Find Maximum

let numbers = [45, 23, 89, 12, 56, 78];
let max = numbers[0];

for (let num of numbers) {
  if (num > max) {
    max = num;
  }
}

console.log("Maximum: " + max);  // 89

// Or use built-in
console.log(Math.max(...numbers));  // 89

Exercise 3: Remove Duplicates

let items = [1, 2, 2, 3, 3, 3, 4];
let unique = [...new Set(items)];

console.log(unique);  // [1, 2, 3, 4]
