Hands-on Tasks

Task 1: User Profile

let user = {
  firstName: "Gokul",
  lastName: "Vinod",
  age: 20,
  email: "gokul@example.com",
  getFullName: function() {
    return this.firstName + " " + this.lastName;
  }
};

console.log(user.getFullName());
console.log("Email: " + user.email);

Task 2: Product Object

  name: "Laptop",
  price: 50000,
  stock: 15,
