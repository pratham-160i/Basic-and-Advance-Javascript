Exercise 2: Car Object

let car = {
  brand: "Toyota",
  model: "Fortuner",
  year: 2022,
  mileage: 0,
  drive: function(km) {
    this.mileage += km;
    return "Drove " + km + " km";
  },
  getInfo: function() {
    return this.year + " " + this.brand + " " + this.model +
           " - Mileage: " + this.mileage + " km";
  }
};

console.log(car.drive(100));
console.log(car.getInfo());

Exercise 3: Object Methods

let book = {
  title: "JavaScript Basics",
  author: "John Doe",
  pages: 350,
  isAvailable: true
};

// Get all keys
console.log(Object.keys(book));

// Get all values
console.log(Object.values(book));

// Check property exists
console.log("author" in book);  // true
